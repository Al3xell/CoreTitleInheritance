designate_core_button = {
    scope = title
    
    is_shown = {
        is_title_core = no
    }
    
    is_valid = {
        can_core_title = yes
    }
    
    effect = {
        # Generate tooltip
        if = {
            limit = {can_core_title = yes}
            custom_tooltip = TITLE_DESIGNATE_CORE_TOOLTIP
        }
        else_if = {
            limit = {is_title_eligible_for_core = no}
            custom_tooltip = TITLE_DESIGNATE_CORE_TOOLTIP_NO_CHILD
        }
        
        # do the coring
        designate_core_title
    }
}
